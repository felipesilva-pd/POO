import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

public class AutorTest {

    @Test
    public void testCriacaoAutor() {
        Autor autor = new Autor("Clarice Lispector", "Brasileira");
        assertEquals("Clarice Lispector", autor.getNome());
        assertEquals("Brasileira", autor.getNacionalidade());
    }

    @Test
    public void testAdicionarLivro() {
        Autor autor = new Autor("Machado de Assis", "Brasileiro");
        Livro livro = new Livro("Dom Casmurro", autor, "Romance", true);
        assertEquals(1, autor.getLivros().size());
        assertEquals("Dom Casmurro", autor.getLivros().get(0).getTitulo());
    }
}
